435  	    RFC_ParamData_t param;

437  	    int len = 0;
438  	
439  	    if (WDMP_SUCCESS == getRFCParameter("PwrMgr", key, &param))
440  	    {
441  	        len = strlen(param.value);
442  	        if (len > MAX_RFC_LEN)
443  	        {
444  	            len = MAX_RFC_LEN;
445  	        }
446  	
447  	        if ( (param.value[0] == '"') && (param.value[len] == '"'))
448  	        {
449  	            strncpy (rfcVal, &param.value[1], len - 1);
450  	            rfcVal[len] = '\0';
451  	        }
452  	        else
453  	        {
454  	            strncpy (rfcVal, param.value, MAX_RFC_LEN-1);
455  	            rfcVal[len] = '\0';
456  	        }
457  	        return atoi(rfcVal);
458  	    }
459  	
460  	    return -1;
461  	}

847  	#ifdef ENABLE_DEEP_SLEEP

905  	void performReboot(const char * requestor, const char * reboot_reason_custom, const char * reboot_reason_other)
906  	{
907  	    LOG("performReboot: Rebooting box now. Requestor: %s. Reboot reason: %s\n", requestor, reboot_reason_custom);
908  	    const char * default_arg = "unknown";
909  	
910  	    check_payload(&requestor, default_arg);
911  	    check_payload(&reboot_reason_custom, default_arg);
912  	    check_payload(&reboot_reason_other, default_arg);
913  	
914  	    IARM_Bus_PWRMgr_RebootParam_t eventData;
915  	    //Note: assumes caller has checked arg reboot_reason_custom and is safe to use.
916  	    strncpy(eventData.reboot_reason_custom, reboot_reason_custom, sizeof(eventData.reboot_reason_custom));
917  	    strncpy(eventData.reboot_reason_other, reboot_reason_other, sizeof(eventData.reboot_reason_other));
918  	    strncpy(eventData.requestor, requestor, sizeof(eventData.requestor));
919  	    eventData.reboot_reason_custom[sizeof(eventData.reboot_reason_custom) - 1] = '\0';
920  	    eventData.reboot_reason_other[sizeof(eventData.reboot_reason_other) - 1] = '\0';
921  	    eventData.requestor[sizeof(eventData.requestor) - 1] = '\0';
922  	    IARM_Bus_BroadcastEvent( IARM_BUS_PWRMGR_NAME, IARM_BUS_PWRMGR_EVENT_REBOOTING, (void *)&eventData, sizeof(eventData));
923  	
924  	    /*
925  	     * performReboot() can be called from the context of an RPC, and the sleep() call below can trigger RPC timeouts. So the time-consuming operations
926  	     * must be handled asynchronously to let this function return promptly. Make a copy of the necessary members so that they can be accessed
927  	     * safely from the new thread.
928  	     * Note: not using strndup() here as lengths of the incoming string are already sanitized.
929  	    */
930  	    char * requestor_cpy = strdup(requestor);
931  	    char * reboot_reason_custom_cpy = strdup(reboot_reason_custom);
932  	    char * reboot_reason_other_cpy = strdup(reboot_reason_other);
933  	
934  	    std::thread async_reboot_thread([requestor_cpy, reboot_reason_custom_cpy, reboot_reason_other_cpy] () {
935  	        v_secure_system("echo 0 > /opt/.rebootFlag");
936  	        sleep(5);
937  	        if(0 == access("/rebootNow.sh", F_OK))
938  	        {
939  	            v_secure_system("/rebootNow.sh -s '%s' -r '%s' -o '%s'", requestor_cpy, reboot_reason_custom_cpy, reboot_reason_other_cpy);
940  	        }
941  	        else
942  	        {
943  	            v_secure_system("/lib/rdk/rebootNow.sh -s '%s' -r '%s' -o '%s'", requestor_cpy, reboot_reason_custom_cpy, reboot_reason_other_cpy);
944  	        }
945  	        free(requestor_cpy);
946  	        free(reboot_reason_custom_cpy);
947  	        free(reboot_reason_other_cpy);
948  	    });
949  	    async_reboot_thread.detach();
950  	}

1755 	{
1756 	    time_t now,wakeup;
1757 	    struct tm wakeupTime;
1758 	    uint32_t wakeupTimeInSec = 0,getTZDiffTime = 0;
1759 	    uint32_t wakeupTimeInMin = 5;
1760 	    FILE *fpWT = NULL;
1761 	
1762 	    /* Read the wakeup Time in Seconds from /tmp override
1763 	       else calculate the Wakeup time til 2AM */
1764 	    fpWT = fopen("/tmp/deepSleepWakeupTimer","r");
1765 	    if (NULL != fpWT)
1766 	    {
1767 	        if(0 > fscanf(fpWT,"%d",&wakeupTimeInMin))
1768 	        {
1769 	            __TIMESTAMP();
1770 	            LOG("Error: fscanf on wakeupTimeInSec failed");
1771 	        }
1772 	        else
1773 	        {
1774 	            wakeupTimeInSec = wakeupTimeInMin * 60 ;
1775 	            fclose (fpWT);
1776 	            __TIMESTAMP();
1777 	            LOG(" /tmp/ override Deep Sleep Wakeup Time is %d \r\n",wakeupTimeInSec);
1778 	            return wakeupTimeInSec;
1779 	        }
1780 	        fclose (fpWT);
1781 	    }
1782 	
1783 	    /* curr time */
1784 	    time (&now);
1785 	
1786 	    /* wakeup time */
1787 	    time (&wakeup);
1788 	    wakeupTime = *localtime (&wakeup);
1789 	
1790 	    if (wakeupTime.tm_hour >=0 && wakeupTime.tm_hour < 2)
1791 	    {
1792 	        /*Calculate the wakeup time till 2 AM..*/
1793 	        wakeupTime.tm_hour = 2;
1794 	        wakeupTime.tm_min = 0;
1795 	        wakeupTime.tm_sec = 0;
1796 	        wakeupTimeInSec = difftime(mktime(&wakeupTime),now);
1797 	
1798 	    }
1799 	    else
1800 	    {
1801 	        /*Calculate the wakeup time till midnight + 2 hours for 2 AM..*/
1802 	        wakeupTime.tm_hour = 23;
1803 	        wakeupTime.tm_min = 59;
1804 	        wakeupTime.tm_sec = 60;
1805 	        wakeupTimeInSec = difftime(mktime(&wakeupTime),now);
1806 	        wakeupTimeInSec = wakeupTimeInSec + 7200; // 7200sec for 2 hours
1807 	    }
1808 	
1809 	    /* Add randomness to calculated value i.e between 2AM - 3AM
1810 	        for 1 hour window
1811 	    */

1823 	    }

1825 	    if ((int64_t)wakeupTimeInSec > INT64_MAX - (int64_t)randTimeInSec) {

1827 	    } else {
1828 	        wakeupTimeInSec += randTimeInSec;
1829 	    }
