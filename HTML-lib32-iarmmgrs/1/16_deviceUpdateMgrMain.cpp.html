181  	IARM_Result_t deviceUpdateStart()
182  	{
183  	
184  		IARM_Result_t status = IARM_RESULT_INVALID_STATE;
185  	
186  		INT_LOG("Entering [%s] - [%s] - disabling io redirect buf\n", __FUNCTION__, IARM_BUS_DEVICE_UPDATE_NAME);
187  		setvbuf(stdout, NULL, _IOLBF, 0);
188  		int ret = pthread_mutex_init(&mapMutex, NULL);
189  	        if(ret != 0) {
190  	        INT_LOG(" pthread_mutex_init Error case: %d\n", __LINE__);
191  	        status = IARM_RESULT_INVALID_STATE;  
192  	        }
193  		if (!initialized)
194  		{
195  			IARM_Result_t rc;
196  	
197  			int retval = pthread_mutex_init(&tMutexLock, NULL);
198  	                if(retval != 0) {
199  	                INT_LOG(" pthread_mutex_init Error case: %d\n", __LINE__);
200  	                }
201  			pthread_mutex_lock(&tMutexLock);
202  			rc = IARM_Bus_Init(IARM_BUS_DEVICE_UPDATE_NAME);
203  			INT_LOG("dumMgr:I-ARM IARM_Bus_Init Mgr: %d\n", rc);
204  	
205  			rc = IARM_Bus_Connect();
206  			INT_LOG("dumMgr:I-ARM IARM_Bus_Connect Mgr: %d\n", rc);
207  	
208  			rc = IARM_Bus_RegisterEvent(IARM_BUS_DEVICE_UPDATE_EVENT_MAX);
209  			INT_LOG("dumMgr:I-ARM IARM_Bus_RegisterEvent Mgr: %d\n", rc);
210  	
211  			rc = IARM_Bus_RegisterCall( IARM_BUS_DEVICE_UPDATE_API_AcceptUpdate, AcceptUpdate); /* RPC Method Implementation*/
212  			INT_LOG("dumMgr:I-ARM IARM_BUS_DEVICE_UPDATE_API_AcceptUpdate Mgr: %d\n", rc);
213  	
214  			IARM_Bus_RegisterEventHandler(IARM_BUS_DEVICE_UPDATE_NAME, IARM_BUS_DEVICE_UPDATE_EVENT_READY_TO_DOWNLOAD,
215  					_deviceUpdateEventHandler);
216  			IARM_Bus_RegisterEventHandler(IARM_BUS_DEVICE_UPDATE_NAME, IARM_BUS_DEVICE_UPDATE_EVENT_DOWNLOAD_STATUS,
217  					_deviceUpdateEventHandler);
218  			IARM_Bus_RegisterEventHandler(IARM_BUS_DEVICE_UPDATE_NAME, IARM_BUS_DEVICE_UPDATE_EVENT_LOAD_STATUS,
219  					_deviceUpdateEventHandler);
220  			IARM_Bus_RegisterEventHandler(IARM_BUS_DEVICE_UPDATE_NAME, IARM_BUS_DEVICE_UPDATE_EVENT_ERROR,
221  					_deviceUpdateEventHandler);
222  	
223  			initialized = 1;
224  	
225  			pthread_mutex_unlock(&tMutexLock);
226  			status = IARM_RESULT_SUCCESS;
227  		}
228  		else
229  		{
230  			__TIMESTAMP();
231  			INT_LOG("dumMgr: I-ARM Device Update Mgr Error case: %d\n", __LINE__);
232  			status = IARM_RESULT_INVALID_STATE;
233  		}
234  		return status;
235  	}

502  	void processDeviceFolder(string updatePath, string deviceName)
503  	{
504  		DIR *dp;
505  		struct dirent *dirp;
506  		INT_LOG("dumMgr:processing device folder:%s\n", updatePath.c_str());
507  		if ((dp = opendir(updatePath.c_str())) == NULL)
508  		{
509  			INT_LOG("dumMgr:Error(%d) opening updatePath \n", errno);
510  			return;
511  		}
512  	
513  	// loop through dir listing and get normal files
514  		INT_LOG("dumMgr:checking folder files\n");
515  		while ((dirp = readdir(dp)) != NULL)
516  		{
517  			INT_LOG("dumMgr:checking folder files:%s  - type=%d\n",dirp->d_name,dirp->d_type);
518  			if (dirp->d_type != DT_DIR)
519  			{
520  				// check for it being either a tgz or tar.gz file
521  				string filename = dirp->d_name;
522  				INT_LOG("dumMgr:checking file:%s\n", filename.c_str());
523  				unsigned int idx = filename.rfind('.');
524  				if (idx != string::npos)
525  				{
526  					string ext = filename.substr(idx + 1);
527  					if (ext == "gz")
528  					{
529  						idx = filename.rfind('.', idx);
530  						if (idx == string::npos)
531  						{
532  							continue;
533  						}
534  						string ext = filename.substr(idx + 1);
535  						if (ext == "tar.gz")
536  						{
537  							processDeviceFile(updatePath + "/" + filename, deviceName);
538  							continue;
539  						}
540  					}
541  					if (ext == "tgz")
542  					{
543  						processDeviceFile(updatePath + "/" + filename, deviceName);
544  						continue;
545  					}
546  				}
547  	
548  			}
549  		}
550  		closedir(dp);
551  	}

553  	void processDeviceFile(string filePath, string deviceName)
554  	{
555  		INT_LOG("dumMgr:processing Device File:%s\n", filePath.c_str());
556  		string cmd = "rm -rf ";
557  		cmd += tempFilePath;
558  	#ifdef YOCTO_BUILD
559  		v_secure_system(cmd.c_str());
560  	#else
561  		system(cmd.c_str());
562  	#endif
563  		cmd = "mkdir ";
564  		cmd += tempFilePath;
565  	#ifdef YOCTO_BUILD
566  	        v_secure_system(cmd.c_str());
567  	#else
568  	        system(cmd.c_str());
569  	#endif
570  		cmd = "tar -xzpf  " + filePath + " -C " + tempFilePath;
571  	#ifdef YOCTO_BUILD
572  	        v_secure_system(cmd.c_str());
573  	#else
574  	        system(cmd.c_str());
575  	#endif
576  		vector<string> *myfiles = getdir(tempFilePath);
577  		vector<string>::iterator itr;
578  		if(myfiles != NULL)
579  		{
580  			INT_LOG("dumMgr: searching Device File:%s\n", filePath.c_str());
581  			for (itr = myfiles->begin(); itr != myfiles->end(); itr++)
582  			{
583  				string filename = *itr;
584  		//		LOG("dumMgr:found file:%s\n", filename.c_str());
585  				int idx = filename.rfind('.');
586  				if (idx == string::npos)
587  				{
588  					continue;
589  				}
590  				string ext = filename.substr(idx + 1);
591  		//		LOG("dumMgr:extension:%s\n", ext.c_str());
592  				if (ext == "xml")

626  	string getXMLTagText(string xml, string tag)
627  	{
628  	
629  	//TODO currently this assume no spaces or tabs in the tag brackets. and no leading trailing spaces in text content
630  		int idx = xml.find("<" + tag);
631  		if (idx == string::npos)
632  		{
633  			INT_LOG("dumMgr:tag <%s> not found in xml file: aborting\n", tag.c_str());
634  			return "";
635  		}
636  	// skip past the tag and its two brackets:
637  		idx += tag.length() + 2;
638  	
639  	// find end tag
640  		int idx2 = xml.find("</" + tag);
641  	
642  	//grab all content between start and end tag
643  		return xml.substr(idx, idx2 - idx);

651  		std::ifstream myfile;

662  			{
663  				return false;
664  			}

696  		myData->forceUpdate = forceUpdate;

698  		return true;

872  		if (strcmp(owner, IARM_BUS_DEVICE_UPDATE_NAME) == 0)
873  		{
874  			pthread_mutex_lock(&tMutexLock);
875

877  			{

879  			{

881  				INT_LOG("I-ARM: got event IARM_BUS_DEVICE_UPDATE_EVENT_READY_TO_DOWNLOAD\n");
882

887  					INT_LOG("I-ARM: got valid updateSessionID  Details:\n");
888  					INT_LOG("I-ARM: 			currentSWVersion:%s\n", eventData->deviceCurrentSWVersion);

895  	
896  					if(interactiveDownload==false){
897  						sendDownLoadInit(eventData->updateSessionID);
898  					}else
899  					{
900  						INT_LOG("I-ARM: 	Interactive Download is true so not sending download Init message\n");
901  					}
902  				}
903  				else
904  				{
905  					INT_LOG("I-ARM: did not get valid updateSessionID  Got id:%d\n", eventData->updateSessionID);
906  				}
907  	
908  			}
909  				break;
910  			case IARM_BUS_DEVICE_UPDATE_EVENT_DOWNLOAD_STATUS:
911  			{
912  				IARM_Bus_DeviceUpdate_DownloadStatus_t *eventData = (IARM_Bus_DeviceUpdate_DownloadStatus_t *) data;
913  				INT_LOG("I-ARM: got event IARM_BUS_DEVICE_UPDATE_EVENT_DOWNLOAD_STATUS\n");
914  				updateInProgress_t *uip = getUpdateInProgress(eventData->updateSessionID);
915

919

921  					if(uip->downloadPercent>=100){
922  						if(interactiveLoad==false){
923  							sendLoadInit(eventData->updateSessionID);
924  						}
925  						else{
926  							INT_LOG("I-ARM: 	Interactive load is true so not sending load Init message\n");
927  						}
928  					}
929  					else{
930  						INT_LOG("I-ARM: 			Not ready to load yet:%i\n", eventData->percentComplete);
931  					}
932  				}
933  				else
934  				{
935  					INT_LOG("I-ARM: IARM_BUS_DEVICE_UPDATE_EVENT_DOWNLOAD_STATUS invalid updateSessionID  Got id:%d\n",

985  			default:
