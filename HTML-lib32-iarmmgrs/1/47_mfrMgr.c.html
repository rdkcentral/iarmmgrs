100  	    IARM_Bus_MFRLib_GetSerializedData_Param_t *param = (IARM_Bus_MFRLib_GetSerializedData_Param_t *)arg;
101  	    mfrError_t err = mfrERR_NONE;
102  	    mfrSerializedData_t data;
103  	    errno_t safec_rc = -1;
104  	    int i;
105  	    if (PROFILE_INVALID == profileType){
106  	        profileType = searchRdkProfile();
107  	    }
108  	    if((param->type == mfrSERIALIZED_TYPE_PROVISIONED_MODELNAME) &&
109  	          (PROFILE_STB == profileType)){
110  	        LOG(" Querying for sky model name ");
111  	        err = mfrGetSerializedData((mfrSerializedType_t)(mfrSERIALIZED_TYPE_SKYMODELNAME), &(data));
112  	    } else {
113  	         err = mfrGetSerializedData((mfrSerializedType_t)(param->type), &(data));
114  	    }
115  	    if(mfrERR_NONE == err)
116  	    {
117  		safec_rc = memcpy_s(param->buffer, sizeof(param->buffer), data.buf, data.bufLen);
118  	    	if(safec_rc != EOK)
119  	        {
120  	                ERR_CHK(safec_rc);
121  	                if(data.freeBuf)
122  	                {
123  	                    data.freeBuf(data.buf);
124  	                }
125  	                return IARM_RESULT_INVALID_PARAM;
126  	         }
127  	         param->bufLen = data.bufLen;
128  	      
129  		if(data.freeBuf)
130  	        {
131  	             data.freeBuf(data.buf);
132  	        }
133  		retCode=IARM_RESULT_SUCCESS;
134  	    }
135  	    return retCode;
