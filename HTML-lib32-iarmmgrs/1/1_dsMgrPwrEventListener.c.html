105  	void initPwrEventListner()
106  	{
107  	    INT_INFO("Entering [%s]\r\n", __FUNCTION__);
108  	    
109  	    profileType = searchRdkProfile();
110  	
111  	    if (PROFILE_TV == profileType )
112  	    {
113  	        if(true == dsMgrProductTraits::ux_controller::initialize_ux_controller(dsMgrProductTraits::DEFAULT_TV_PROFILE))
114  	        {
115  	            ux = dsMgrProductTraits::ux_controller::get_instance();
116  	        }
117  	    }
118  	    else
119  	    {
120  	        if(true == dsMgrProductTraits::ux_controller::initialize_ux_controller(dsMgrProductTraits::DEFAULT_STB_PROFILE_EUROPE))
121  	        {
122  	            ux = dsMgrProductTraits::ux_controller::get_instance();
123  	        }
124  	    }
125  	    if(nullptr == ux)
126  	    {
127  	        INT_DEBUG("dsmgr product traits not supported.\n");
128  	    }
129  	
130  	    try {
131  	        device::Manager::load();
132  	        INT_DEBUG("device::Manager::load success.\n");
133  	    }
134  	    catch (...){
135  	        INT_DEBUG("Exception Caught during [device::Manager::load]\r\n");
136  	    }   
137  	
138  	    IARM_Bus_RegisterCall(IARM_BUS_DSMGR_API_SetStandbyVideoState, _SetStandbyVideoState);
139  	    IARM_Bus_RegisterCall(IARM_BUS_DSMGR_API_GetStandbyVideoState, _GetStandbyVideoState);
140  	    IARM_Bus_RegisterCall(IARM_BUS_DSMGR_API_SetAvPortState, _SetAvPortState);
141  	    IARM_Bus_RegisterCall(IARM_BUS_DSMGR_API_SetLEDStatus, _SetLEDState);
142  	    IARM_Bus_RegisterCall(IARM_BUS_DSMGR_API_SetRebootConfig, _SetRebootConfig);
143  	
144  	    /* Check for Power Controller Connection and if failure, start a new thread and wait until connection established 
145  	     * if connection is eastablished, proceed with fetching and Getting values from Power control and initialize*/
146  	    if(POWER_CONTROLLER_ERROR_NONE == PowerController_Connect())
147  	    {
148  	        INT_DEBUG("PowerController_Connect is Success \r\n");
149  	        dsMgrPwrControllerFetchNinitStateValues();
150  	    }
151  	    else
152  	    {
153  	        INT_ERROR("DSMgr PowerController_Connect Failed Retrying in Thread\r\n");
154  	        dsMgrPwrCtrlEstablishConnection();
155  	    }
156  	}

158  	static void dsMgrPwrCtrlEstablishConnection(void)
159  	{
160  	    INT_INFO("Entering [%s]\r\n", __FUNCTION__);
161  	    
162  	    pthread_t edsPwrConnectThreadID; 
163  	
164  	    if(pthread_create (&edsPwrConnectThreadID, NULL, dsMgrPwrRetryEstablishConnThread, NULL)  == 0)
165  	    {
166  	        if(pthread_detach(edsPwrConnectThreadID) != 0) 
167  	        {
168  	            INT_ERROR("DSMgr dsMgrPwrCtrlEstablishConnection Thread detach Failed\r\n");
169  	        }
170  	    }
171  	    else
172  	    {
173  	        INT_ERROR("DSMgr dsMgrPwrCtrlEstablishConnection Thread Creation Failed\r\n");
174  	    }
175  	}
176  	
177  	static void* dsMgrPwrRetryEstablishConnThread(void *arg)
178  	{
179  	    INT_INFO("%s: Entry  \r\n",__FUNCTION__);
180  	
181  	    while(1)
182  	    {
183  	        if(POWER_CONTROLLER_ERROR_NONE == PowerController_Connect())
184  	        {
185  	            INT_DEBUG("dsMgrPwrRetryEstablishConnThread PowerController_Connect is Success \r\n");
186  	            dsMgrPwrControllerFetchNinitStateValues();
187  	            break;
188  	        }
189  	        else
190  	        {
191  	            /* 300 msec wait and recheck connect */
192  	            usleep(DSMGR_PWR_CNTRL_CONNECT_WAIT_TIME_MS);
193  	        }
194  	    }
195  	    INT_DEBUG("dsMgrPwrRetryEstablishConnThread Completed Exit \r\n");
196  	    return arg;
197  	}

199  	static void dsMgrPwrControllerFetchNinitStateValues(void)
200  	{
201  	    INT_INFO("Entering [%s]\r\n", __FUNCTION__);
202  	    
203  	    PowerController_PowerState_t _curState = POWER_STATE_UNKNOWN;
204  	    PowerController_PowerState_t _prevState = POWER_STATE_UNKNOWN;
205  	    PowerController_PowerState_t powerStateBeforeReboot = POWER_STATE_STANDBY;
206  	
207  	    PowerController_RegisterPowerModeChangedCallback(_PwrEventHandler, nullptr);
208  	    /*  Read the Device Power State on startup... */
209  	    if (POWER_CONTROLLER_ERROR_NONE == PowerController_GetPowerState(&_curState, &_prevState))
210  	    {
211  	        INT_DEBUG("Deep Sleep Manager Init with Power State %d\r\n", _curState);
212  	        curState = _curState;
213  	    }
214  	
215  	    if (POWER_CONTROLLER_ERROR_NONE != PowerController_GetPowerStateBeforeReboot(&powerStateBeforeReboot))
216  	    {
217  	        INT_ERROR("DSMgr GetPowerStateBeforeReboot Failed\r\n");
218  	    }
219  	
220  	    if (nullptr != ux) 
221  	    {
222  	        // This will set up ports, lights and bootloader pattern internally
223  	        ux->applyPostRebootConfig(curState, powerStateBeforeReboot);
224  	    } else {
225  	        // Sync Port with Power state TODO:
226  	    }
227  	
228  	    if (nullptr == ux) // Since ux_controller is not supported, ports need to be
229  	                       // set up explicitly.
230  	    {
231  	#ifndef DISABLE_LED_SYNC_IN_BOOTUP
232  	        _SetLEDStatus(curState);
233  	#endif
234  	        _SetAVPortsPowerState(curState);
235  	    }
236  	}

238  	static void _PwrEventHandler(const PowerController_PowerState_t currentState,
239  	    const PowerController_PowerState_t newState,
240  	    void* userdata)
241  	{
242  	    INT_INFO("Entering [%s]\r\n", __FUNCTION__);
243  	
244  	    /* In this call back function, received event is pushed to queue guarded by queue mutex and signaled for the thread to 
245  	        pop out from queue and process the pushed event. This will allow the CB to be handled separately 
246  	        in a different thread context, userdata parameter is empty and not handled*/
247  	
248  	    pthread_mutex_lock(&tdsPwrEventQueueMutexLock);
249  	    pwrEventQueue.emplace(currentState, newState);
250  	    pthread_mutex_unlock(&tdsPwrEventQueueMutexLock);
251  	    
252  	    INT_DEBUG("Sending Signal to Thread for Processing Callback Event \r\n");
253  	
254  	    pthread_mutex_lock(&tdsPwrEventMutexLock);
255  	    pthread_cond_signal(&tdsPwrEventMutexCond);
256  	    pthread_mutex_unlock(&tdsPwrEventMutexLock);
257  	}

259  	static int _SetLEDStatus(PowerController_PowerState_t powerState)
260  	{
261  	    INT_INFO("Entering [%s]\r\n", __FUNCTION__);
262  	    try {
263  	        
264  	        dsFPDStateParam_t param;
265  	
266  	        param.eIndicator = dsFPD_INDICATOR_POWER;
267  	
268  	        if( POWER_STATE_ON  != powerState )
269  	        {
270  	            if(PROFILE_TV == profileType)
271  	            {
272  	                param.state = dsFPD_STATE_ON;
273  	                INT_DEBUG("[PWRMgr-_SetLEDStatus] Settings the Power LED State to ON\r\n");
274  	            }
275  	            else
276  	            {
277  	                param.state = dsFPD_STATE_OFF;
278  	                INT_DEBUG("[PWRMgr-_SetLEDStatus] Settings the Power LED State to OFF \r\n");
279  	            }
280  	        }
281  	        else
282  	        {
283  	            param.state = dsFPD_STATE_ON;
284  	            INT_DEBUG("[PWRMgr-_SetLEDStatus] Settings the Power LED State to ON \r\n");
285  	        }
286  	
287  	        _dsSetFPState(&param);
288  	    }
289  	    catch (...) {
290  	        INT_DEBUG("Exception Caught during [PWRMgr - _SetLEDStatus]\r\n");
291  	        return 0;
292  	    }
293  	    return 0;

296  	int _SetAVPortsPowerState(PowerController_PowerState_t powerState)
297  	{
298  	    INT_INFO("Entering [%s] powerState:%d \r\n", __FUNCTION__,powerState);
299  	
300  	    try
301  	    {
302  	        if (POWER_STATE_ON != powerState)
303  	        {
304  	            if (POWER_STATE_OFF != powerState)
305  	            {
306  	                INT_INFO("[%s] POWERSTATE %d \r\n", __FUNCTION__, powerState);
307  	                // We're in one of the standby modes. Certain ports may have to be left on.
308  	                try
309  	                {
310  	                    device::List<device::VideoOutputPort> videoPorts = device::Host::getInstance().getVideoOutputPorts();
311  	                    for (size_t i = 0; i < videoPorts.size(); i++)
312  	                    {
313  	                        bool doEnable = get_video_port_standby_setting(videoPorts.at(i).getName().c_str());
314  	                        INT_DEBUG("[%s]Video port %s will be %s in standby mode.\n", __FUNCTION__, videoPorts.at(i).getName().c_str(), (doEnable ? "enabled" : "disabled"));
315  	                        if (false == doEnable)
316  	                            videoPorts.at(i).disable();
317  	                    }
318  	                }
319  	                catch (...)
320  	                {
321  	                    INT_DEBUG("[%s] Video port exception %d \r\n", __FUNCTION__, powerState);
322  	                }
323  	                INT_INFO("[%s] Video port standby done \r\n",__FUNCTION__);
324  	            }
325  	            else
326  	            {
327  	                try
328  	                {
329  	                    // Disable all ports when going into POWERSTATE_OFF
330  	                    device::List<device::VideoOutputPort> videoPorts = device::Host::getInstance().getVideoOutputPorts();
331  	                    for (size_t i = 0; i < videoPorts.size(); i++)
332  	                    {
333  	                        videoPorts.at(i).disable();
334  	                    }
335  	                }
336  	                catch (...)
337  	                {
338  	                    INT_DEBUG("[%s] Video port exception %d \r\n", __FUNCTION__, powerState);
339  	                }
340  	                INT_INFO("[%s] Video port disable done \r\n", __FUNCTION__ );
341  	            }
342  	            try
343  	            {
344  	                dsAudioGetHandleParam_t getHandle;
345  	                dsAudioPortEnabledParam_t setMode;
346  	                int numPorts, i = 0;
347  	
348  	                numPorts = dsUTL_DIM(kSupportedPortTypes);
349  	                for (i = 0; i < numPorts; i++)
350  	                {
351  	                    const dsAudioPortType_t *audioPort = &kSupportedPortTypes[i];
352  			    device::AudioOutputPort aPort = device::Host::getInstance().getAudioOutputPort(*audioPort);
353  	                    memset(&getHandle, 0, sizeof(getHandle));
354  	                    getHandle.type = *audioPort;
355  	                    getHandle.index = 0;
356  	                    _dsGetAudioPort(&getHandle);
357  	
358  	                    memset(&setMode, 0, sizeof(setMode));
359  	                    setMode.handle = getHandle.handle;
360  			    strncpy(setMode.portName, aPort.getName().c_str(), 31);
361  	                    setMode.enabled = false;
362  	                    _dsEnableAudioPort(&setMode);
363  	                }
364  	            }
365  	            catch (...)
366  	            {
367  	                INT_DEBUG("[%s] audio port exception \r\n", __FUNCTION__);
368  	            }
369  	        }
370  	        else
371  	        {
372  	
373  	            try
374  	            {
375  	                device::List<device::VideoOutputPort> videoPorts = device::Host::getInstance().getVideoOutputPorts();
376  	                for (size_t i = 0; i < videoPorts.size(); i++)
377  	                {
378  	                    videoPorts.at(i).enable();
379  	                }
380  	        
381  	                dsAudioGetHandleParam_t getHandle;
382  	                dsAudioPortEnabledParam_t setMode;
383  	
384  	                int numPorts, i = 0;
385  	
386  	                numPorts = dsUTL_DIM(kSupportedPortTypes);
387  	                for (i = 0; i < numPorts; i++)
388  	                {
389  	                    const dsAudioPortType_t *audioPort = &kSupportedPortTypes[i];
390  			    device::AudioOutputPort aPort = device::Host::getInstance().getAudioOutputPort(*audioPort);
391  	                    memset(&getHandle, 0, sizeof(getHandle));
392  	                    getHandle.type = *audioPort;
393  	                    getHandle.index = 0;
394  	                    _dsGetAudioPort(&getHandle);
395  	
396  	                    memset(&setMode, 0, sizeof(setMode));
397  	                    setMode.handle = getHandle.handle;
398  			    strncpy(setMode.portName, aPort.getName().c_str(), 31);
399  	                    setMode.enabled = false;
400  	                    _dsGetEnablePersist(&setMode);
401  	
402  	                    if (setMode.enabled == true)
403  	                    {
404  	                        /*Instead of enabling all the audio ports on power transition */
405  	                        /*Get the values from persistent storage & update */
406  	                        INT_INFO("[%s] Enabling audio ports %d \r\n", __FUNCTION__, powerState);
407  	                        _dsEnableAudioPort(&setMode);
408  	                    }
409  	                }
410  	                if (isEAS == IARM_BUS_SYS_MODE_EAS)
411  	                {
412  	                    /* Force Stereo in EAS mode. */
413  	                    INT_INFO("[%s] Force Stereo in EAS mode \r\n", __FUNCTION__);
414  	                    _setEASAudioMode();
415  	                }
416  	            }
417  	            catch (...)
418  	            {
419  	                INT_DEBUG("[%s] Audio port exception \r\n",__FUNCTION__);
420  	            }
421  	
422  	        }
423  	    }
424  	    catch (...)
425  	    {
426  	        INT_DEBUG("Exception Caught during [%s]\r\n", __FUNCTION__);
427  	        return 0;
428  	    }
429  	    return 0;
430  	}

432  	static bool get_video_port_standby_setting(const char * port)
433  	{
434  	    INT_INFO("Entering [%s]\r\n", __FUNCTION__);
435  	    if(NULL == port)
436  	    {
437  	        INT_DEBUG("[%s]Error! Port name is NULL!\n", __FUNCTION__);
438  	        return false;
439  	    }
440  	    for(int i = 0; i < MAX_NUM_VIDEO_PORTS; i++)
441  	    {
442  	        if(0 == strncasecmp(port, g_standby_video_port_setting[i].port, DSMGR_MAX_VIDEO_PORT_NAME_LENGTH))
443  	        {
444  	            return g_standby_video_port_setting[i].isEnabled;
445  	        }
446  	    }
447  	    return false; //Default setting: video port is disabled in standby mode
448  	}

450  	static IARM_Result_t _SetStandbyVideoState(void *arg)
451  	{
452  	    INT_INFO("Entering [%s]\r\n", __FUNCTION__);
453  	    dsMgrStandbyVideoStateParam_t *param = (dsMgrStandbyVideoStateParam_t *)arg;
454  	    if(NULL == param->port)
455  	    {
456  	        param->result = -1;
457  	        INT_DEBUG("[%s] empty port name. Cannot proceed.\n", __FUNCTION__);
458  	        return IARM_RESULT_SUCCESS;
459  	    }
460  	    else
461  	        param->result = 0;
462  	
463  	    int i = 0;
464  	    for(i = 0; i < MAX_NUM_VIDEO_PORTS; i++)
465  	    {
466  	        if(0 == strncasecmp(param->port, g_standby_video_port_setting[i].port, DSMGR_MAX_VIDEO_PORT_NAME_LENGTH))
467  	        {
468  	            /*Found a match. Update it*/
469  	            g_standby_video_port_setting[i].isEnabled = ((0 == param->isEnabled) ? false : true);
470  	            break;
471  	        }
472  	    }
473  	    if(MAX_NUM_VIDEO_PORTS == i)
474  	    {
475  	        /*No matching entries are present. Add one.*/
476  	        for(i = 0; i < MAX_NUM_VIDEO_PORTS; i++)
477  	        {
478  	            if('\0' == g_standby_video_port_setting[i].port[0])
479  	            {
480  	                strncpy(g_standby_video_port_setting[i].port, param->port, (DSMGR_MAX_VIDEO_PORT_NAME_LENGTH - 1));
481  	                g_standby_video_port_setting[i].isEnabled = ((0 == param->isEnabled) ? false : true);
482  	                break;
483  	            }
484  	        }
485  	    }
486  	    if(MAX_NUM_VIDEO_PORTS == i)
487  	    {
488  	        INT_DEBUG("[%s] Error! Out of room to write new video port setting for standby mode.\n", __FUNCTION__);
489  	    }
490  	
491  	    try
492  	    {
493  	        device::VideoOutputPort &vPort = device::Host::getInstance().getVideoOutputPort(param->port);
494  	        if((POWER_STATE_ON != curState) && (POWER_STATE_OFF != curState))
495  	        {
496  	            /*We're currently in one of the standby states. This new setting needs to be applied right away.*/
497  	            INT_DEBUG("[%s] Setting standby %s port status to %s.\n", __FUNCTION__, param->port, ((1 == param->isEnabled)? "enabled" : "disabled"));
498  	            if(1 == param->isEnabled)
499  	                vPort.enable();
500  	            else
501  	                vPort.disable();
502  	        }
503  	        else
504  	        {
505  	            INT_DEBUG("[%s] video port %s will be %s when going into standby mode.\n", __FUNCTION__, param->port, ((1 == param->isEnabled)? "enabled" : "disabled"));
506  	        }
507  	    }
508  	    catch (...)
509  	    {
510  	        INT_DEBUG("Exception Caught during [%s]. Possible bad video port.\n", __FUNCTION__);
511  	        param->result = -1;
512  	    }
513  	    return IARM_RESULT_SUCCESS;
514  	}

516  	static IARM_Result_t _GetStandbyVideoState(void *arg)
517  	{
518  	    INT_INFO("Entering [%s]\r\n", __FUNCTION__);
519  	    dsMgrStandbyVideoStateParam_t *param = (dsMgrStandbyVideoStateParam_t *)arg;
520  	    if(NULL == param->port)
521  	    {
522  	        INT_DEBUG("[%s]Bad port name. Cannot get state.\n", __FUNCTION__);
523  	        return IARM_RESULT_SUCCESS;
524  	    }
525  	
526  	    try
527  	    {
528  	        device::VideoOutputPort &vPort = device::Host::getInstance().getVideoOutputPort(param->port);
529  	
530  	    }
531  	    catch (...)
532  	    {
533  	        INT_DEBUG("Exception Caught during [%s]. Possible bad video port.\n", __FUNCTION__);
534  	        param->result = -1;
535  	        return IARM_RESULT_SUCCESS;
536  	    }
537  	    param->isEnabled = (get_video_port_standby_setting(param->port) ? 1 : 0);
538  	    param->result = 0;
539  	    return IARM_RESULT_SUCCESS;
540  	}

542  	static PowerController_PowerState_t dsMgrPwrMgrToPowerControllerPowerState(PWRMgr_PowerState_t _state)
543  	{
544  	    PowerController_PowerState_t powerState = POWER_STATE_UNKNOWN;
545  	    switch (_state) {
546  	    case PWRMGR_POWERSTATE_OFF:
547  	        powerState = POWER_STATE_OFF;
548  	        break;
549  	    case PWRMGR_POWERSTATE_STANDBY:
550  	        powerState = POWER_STATE_STANDBY;
551  	        break;
552  	    case PWRMGR_POWERSTATE_ON:
553  	        powerState = POWER_STATE_ON;
554  	        break;
555  	    case PWRMGR_POWERSTATE_STANDBY_LIGHT_SLEEP:
556  	        powerState = POWER_STATE_STANDBY_LIGHT_SLEEP;
557  	        break;
558  	    case PWRMGR_POWERSTATE_STANDBY_DEEP_SLEEP:
559  	        powerState = POWER_STATE_STANDBY_DEEP_SLEEP;
560  	        break;
561  	    default:
562  	        /* powerState is already UNKNOWN in the init of the fn*/
563  	        INT_ERROR("DSMgr dsMgrPwrMgrToPowerControllerPowerState Invalid Power State\r\n");
564  	        break;
565  	    }
566  	    INT_DEBUG("[%s] _state=[%d]  powerState[%d]\r\n", __FUNCTION__,_state,powerState);
567  	    return powerState;
568  	}

637  	static void* dsMgrPwrEventHandlingThreadFunc(void *arg)
638  	{
639  	    INT_INFO("%s: Entry  \r\n",__FUNCTION__);
640  	
641  	    /* In Loop and waiting for an event conditionally */
642  	    while (true)
643  	    {
(1) Event lock_acquire: Calling "pthread_mutex_lock" acquires lock "tdsPwrEventMutexLock".
Also see events: [loop_outside_lock_insufficient][wait_not_in_locked_loop][remediation]
644  	        pthread_mutex_lock(&tdsPwrEventMutexLock);
645  	        INT_DEBUG("dsMgrPwrEventHandlingThreadFunc.... Wait for Events from Power manager Controller Callback\r\n");
(3) Event wait_not_in_locked_loop: A wait is performed without a loop. If there is a spurious wakeup, the condition may not be satisfied.
(4) Event remediation: Check the wait condition in a loop, with the lock held. The lock must not be released between the condition and the wait.
Also see events: [lock_acquire][loop_outside_lock_insufficient]
646  	        int waitStatus = pthread_cond_wait(&tdsPwrEventMutexCond, &tdsPwrEventMutexLock);
(2) Event loop_outside_lock_insufficient: Check of "0 != waitStatus" is not sufficient as the lock is not held continuously until the wait. The condition could change between the check and the wait.
Also see events: [lock_acquire][wait_not_in_locked_loop][remediation]
647  	        if (0 != waitStatus) {
648  	            pthread_mutex_unlock(&tdsPwrEventMutexLock);
649  	            INT_ERROR("[%s:%d]dsMgrPwrEventHandlingThreadFunc: pthread_cond_wait failed = %u\n", __FUNCTION__, __LINE__, waitStatus);
650  		    /*
651  		     * Coverity note: No shared data is accessed after pthread_cond_wait failure.
652  		     * The mutex is unlocked and the thread exits immediately, so this is safe.
653  		     */
654  	            break;
655  	        }
656  	
657  	        if(m_dsMgrPwrStopThread)
658  	        {
659  	             /* This case can enter if the de init is trigerred which wants to exit the thread function 
660  	                  and unlock happens after the end of while loop, it will not process any events for this case*/
661  	            INT_DEBUG("[%s:%d]:dsMgrPwrEventHandlingThreadFunc Exiting due to m_dsMgrPwrStopThread true \n",__FUNCTION__,__LINE__);
662  	            pthread_mutex_unlock(&tdsPwrEventMutexLock);
663  	            break;
664  	        }
665  	        pthread_mutex_unlock(&tdsPwrEventMutexLock);
666  	
667  	        
668  		/*  Directly read the contents of the PwrEvtQueue, release the lock when processing the data and relock it after 
669  	            wards until the complete data in the queue is processed. This will ensure the lock retention period is less 
670  		    and all the updated data from the queue is processed */
671  		pthread_mutex_lock(&tdsPwrEventQueueMutexLock);
672  		while (!pwrEventQueue.empty())
673  		{
674  	            DSMgr_Power_Event_State_t pwrEvent = pwrEventQueue.front();
675  	            pwrEventQueue.pop();
676  	            pthread_mutex_unlock(&tdsPwrEventQueueMutexLock);
677  	            /* Release the Lock when processing the data */
678  		    dsMgrHandlePwrEventData(pwrEvent.currentState,pwrEvent.newState);
679  	            /* Reacquire the Lock before checking the Queue contents in loop */
680  	            pthread_mutex_lock(&tdsPwrEventQueueMutexLock);
681  	        }
682  	        /* Release the Lock either exit is due to empty directly or processing completed  */
683  	        pthread_mutex_unlock(&tdsPwrEventQueueMutexLock);
684  	    }
685  	    return arg;
686  	}

688  	static void dsMgrHandlePwrEventData(const PowerController_PowerState_t currentState,
689  	    const PowerController_PowerState_t newState)
690  	{
691  	    INT_INFO("Entering [%s]\r\n", __FUNCTION__);
692  	
693  	    /* This is the Handler for Power State after Separation of the Context to a new thread context.
694  	        This function is getting executed in a new thread context.with currentState and newState values*/
695  	    INT_DEBUG("In [%s] currentState [%d]   newState[%d]\r\n", __FUNCTION__,currentState,newState);
696  	    
697  	    if (nullptr != ux) {
698  	        // If ux_controller is supported, it will set up AV ports and LEDs in the below call.
699  	        ux->applyPowerStateChangeConfig(newState, currentState);
700  	    } else {
701  	#ifndef DISABLE_LED_SYNC_IN_BOOTUP
702  	        _SetLEDStatus(newState);
703  	#endif
704  	        _SetAVPortsPowerState(newState);
705  	    }

707  	}

710  	void dsMgrInitPwrControllerEvt(void)
711  	{
712  	    INT_INFO("Entering [%s]\r\n", __FUNCTION__);
713  	
714  	    /* Initialize two Mutexes and Wait Signal Variable. 
715  	     *  1. One Lock mutex variable to Guard the Event Queue 
716  	     *  2. Another mutex variable to Guard the Wait -Signal Condition, thread creation variable,stop thread
717  	     *  Create the thread used to separate the context of the Event Callback 
718  	     */
719  	     
720  	    pthread_mutex_init (&tdsPwrEventQueueMutexLock, NULL);
721  	    pthread_mutex_init (&tdsPwrEventMutexLock, NULL);
722  	    pthread_cond_init (&tdsPwrEventMutexCond, NULL);
723  	
724  	    m_dsMgrPwrStopThread = false;
725  	    if(pthread_create (&edsPwrEventHandlerThreadID, NULL, dsMgrPwrEventHandlingThreadFunc, NULL) != 0)
726  	    {
727  	        INT_ERROR("DSMgr dsMgrInitPwrControllerEvt Thread not created\r\n");
728  	    }
729  	}

731  	void dsMgrDeinitPwrControllerEvt(void) 
732  	{    
733  	    INT_INFO("Entering [%s]\r\n", __FUNCTION__);
734  	
735  	    /* Lock the Queue, Clear the elements by looping through the Queue, Unlock  and Destroy the Mutex Lock  */
736  	    pthread_mutex_lock(&tdsPwrEventMutexLock);
737  	    /* Notify the Event thread function to Exit, send signal with stop thread true so that the thread exits   */
738  	    m_dsMgrPwrStopThread=true;
739  	    pthread_cond_signal(&tdsPwrEventMutexCond);
740  	    pthread_mutex_unlock(&tdsPwrEventMutexLock);
741  	
742  	    /* wait for the event thread handler to complete its operation */
743  	    INT_DEBUG("[%s:%d]:Before Joining thread id\n",__FUNCTION__,__LINE__);
744  	    pthread_join(edsPwrEventHandlerThreadID, NULL);
745  	    INT_DEBUG("[%s:%d]:Completed  Joined thread id\n",__FUNCTION__,__LINE__);
746  	
747  	    /* clean the queue with guarding mutex*/
748  	    pthread_mutex_lock(&tdsPwrEventQueueMutexLock); 
749  	    pwrEventQueue = std::queue<DSMgr_Power_Event_State_t>();
750  	    pthread_mutex_unlock(&tdsPwrEventQueueMutexLock);
751  	
752  	
753  	    /* Lock the Mutex, Destroy the Wait/Signal Condition Variable, Unlock the Mutex*/
754  	    pthread_cond_destroy(&tdsPwrEventMutexCond);
755  	
756  	    /* Destroy mutexes */
757  	    pthread_mutex_destroy(&tdsPwrEventQueueMutexLock);
758  	    pthread_mutex_destroy(&tdsPwrEventMutexLock);
759  	}
