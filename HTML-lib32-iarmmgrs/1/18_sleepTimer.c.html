64   	static gboolean _onSleepTimerFire(void *)
65   	{
66   	    LOCK();
67   	    LOG("SleepTimer %d start at %ld, fired at %ld, set standby\r\n", timerSource, timerEnd, time(NULL));
68   	    timerEnd = 0;
69   	    timerSource = 0;
70   	    UNLOCK();
71   	#ifdef SLEEPTIMER_USE_CECSTANDBY
72   	    SendCECStandby();
73   	#else
74   	    LOG("No STANDBY action implemented\r\n");
75   	#endif
76   	    return FALSE;
77   	}

79   	static IARM_Result_t _SetSleepTimer(void *arg)
80   	{
81   	    IARM_Bus_PWRMgr_SleepTimer_Param_t *param = (IARM_Bus_PWRMgr_SleepTimer_Param_t *)arg;
82   	
83   	    if(param != NULL)
84   	    {
85   	        LOCK();
86   	        if (param->start == 1) {
87   	            time(&timerEnd);
88   	            timerEnd += (time_t)param->time;
89   	            LOG("SleepTimer (re)set :%.3f, curr source=%d, to fire at %ld\r\n", param->time, timerSource, timerEnd);
90   	            if (timerSource) {
91   	                g_source_remove(timerSource);
92   	                timerSource = 0;
93   	            }
94   	            timerSource = g_timeout_add_seconds((guint)(param->time), _onSleepTimerFire, NULL);
95   	        }
96   	        else if (param->start == 0) {
97   	            LOG("SleepTimer %d cancelled\r\n", timerSource);
98   	            if (timerSource) {
99   	                g_source_remove(timerSource);
100  	            }
101  	            timerSource = 0;
102  	            timerEnd = 0;
103  	        }
104  	        UNLOCK();
105  	        return IARM_RESULT_SUCCESS;
106  	    }
107  	    return IARM_RESULT_IPCCORE_FAIL;
108  	}
109  	
110  	static IARM_Result_t _GetSleepTimer(void *arg)
111  	{
112  	    IARM_Bus_PWRMgr_SleepTimer_Param_t *param = (IARM_Bus_PWRMgr_SleepTimer_Param_t *)arg;
113  	    param->time = 0.;
114  	    param->start = 0;
115  	    LOCK();
116  	    if (timerSource) {
117  	        time_t now = time(NULL);
118  	        param->time = (double)((now < timerEnd) ? (timerEnd - now) :0);
119  	        param->start = (param->time ? 1 : 0);
120  	    }
121  	    UNLOCK();
122  	    LOG("SleepTimer %d has %.3lf remaining\r\n", param->start, param->time);  //CID:128053,127752,97127,95718 - Print_Args
123  	    return IARM_RESULT_SUCCESS;
124  	}

126  	void IARM_Bus_PWRMGR_RegisterSleepTimerAPIs(void *context)
127  	{
128  	    LOG("SleepTimer PwrMgr register APIs\r\n");
129  	    mainloop = (GMainLoop*)context;
130  	    IARM_Bus_RegisterCall(IARM_BUS_PWRMGR_API_SetSleepTimer, _SetSleepTimer);
131  	    IARM_Bus_RegisterCall(IARM_BUS_PWRMGR_API_GetSleepTimer, _GetSleepTimer);
