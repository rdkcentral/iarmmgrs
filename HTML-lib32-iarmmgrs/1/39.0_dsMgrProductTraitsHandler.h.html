64   	/*
65   	    ux_controller or 'user experience controller' is in charge of maintaining and applying the user experience attributes owned by power manager.
66   	    That means attributes like this:
67   	     * Preferred power mode when the device reboots.
68   	     * power LED configuration when device is in standby and ON modes.
69   	     * Support for silent reboot (suppressing flash screen, power LED etc) when executing a maintenance reboot.
70   	     * and more.
71   	
72   	    User experiences will vary among the various product classes. The various derivates of ux_controller class will represent each such specialiation.
73   	*/
74   	class ux_controller
75   	{
76   	protected:
77   	    unsigned int id;
78   	    std::string name;
79   	    deviceType_t deviceType;
80   	    bool invalidateAsyncBootloaderPattern;
81   	    bool firstPowerTransitionComplete;
82   	    mutable std::mutex mutex;
83   	
84   	    bool enableMultiColourLedSupport;
85   	
86   	    bool ledEnabledInStandby;

Also see events: [uninit_member][member_decl][uninit_member]
87   	    int ledColorInStandby;
88   	
89   	    bool ledEnabledInOnState;
(3) Event member_decl: Class member declaration for "ledColorInOnState".
Also see events: [member_decl][uninit_member][uninit_member]
90   	    int ledColorInOnState;
91   	
92   	    powerModeTrait_t preferedPowerModeOnReboot;
93   	    bool enableSilentRebootSupport;
94   	
95   	    static ux_controller * singleton;
96   	
97   	    void initialize_safe_defaults();
98   	    void sync_power_led_with_power_state(PowerController_PowerState_t state) const;
99   	    inline void sync_display_ports_with_power_state(PowerController_PowerState_t state) const;
100  	    bool _set_bootloader_pattern(mfrBlPattern_t pattern) const;
101  	    void _set_bootloader_pattern_async(mfrBlPattern_t pattern) const;
102  	    bool set_bootloader_pattern(mfrBlPattern_t pattern);
103  	    bool set_bootloader_pattern_fault_tolerant(mfrBlPattern_t pattern);
104  	
105  	public:
106  	    static bool initialize_ux_controller(unsigned int profile_id); //Not thread-safe
107  	    static ux_controller * get_instance(); //Not thread-safe
108  	    ux_controller(unsigned int in_id, const std::string &in_name, deviceType_t in_device_type);
109  	    virtual ~ux_controller() {}
110  	
111  	    virtual bool applyPowerStateChangeConfig(PowerController_PowerState_t new_state, PowerController_PowerState_t prev_state) {
112  	        return false;
113  	    };
114  	    virtual bool applyPreRebootConfig(PowerController_PowerState_t current_state) const {
115  	        return false;
116  	    }
117  	    virtual bool applyPreMaintenanceRebootConfig(PowerController_PowerState_t current_state) {
118  	        return false;
119  	    }
120  	    virtual bool applyPostRebootConfig(PowerController_PowerState_t new_state, PowerController_PowerState_t prev_state) {
121  	        return false;
122  	    }
123  	    virtual PowerController_PowerState_t getPreferredPostRebootPowerState(PowerController_PowerState_t prev_state) const {
124  	        return prev_state;
125  	    }
126  	    virtual void sync_display_ports_with_reboot_reason(reboot_type_t type) {};
127  	
128  	
129  	};
