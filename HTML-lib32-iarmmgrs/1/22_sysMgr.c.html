170  	IARM_Result_t SYSMgr_Loop()
171  	{
172  	    time_t curr = 0;
173  	    while(1)
174  	    {
175  	        time(&curr);
176  	        LOG("I-ARM Sys Mgr: HeartBeat at %s\r\n", ctime(&curr));
177  	        sleep(2000);
178  	    }
179  	    return IARM_RESULT_SUCCESS;
180  	}

183  	IARM_Result_t SYSMgr_Stop(void)
184  	{
185  	    IARM_Result_t retStatus = IARM_RESULT_INVALID_STATE;
186  	    pthread_mutex_lock(&tMutexLock);
187  	    if (initialized) {
188  	        CHECK_AND_RETURN_ERROR(IARM_Bus_Disconnect());
189  	        CHECK_AND_RETURN_ERROR(IARM_Bus_Term());
190  	        initialized = 0;
191  	        pthread_mutex_unlock(&tMutexLock);
192  	        pthread_mutex_destroy(&tMutexLock);
193  	    }
194  	    else {
195  	        pthread_mutex_unlock(&tMutexLock);
196  	        retStatus = IARM_RESULT_INVALID_STATE;
197  	    }
198  	    return retStatus;
199  	}

202  	 * @brief This functions sets/updates the HDCP Profile
203  	 *
204  	 * @param callCtx: Context to the caller function
205  	 * @param methodID: Method to be invoke
206  	 * @param arg: Specifies the timeout to be set
207  	 * @param serial: Handshake code to share with
208  	 *
209  	 * @return None
210  	 */
211  	static int CheckHdcpProfile(void)
212  	{
213  	    struct stat st;
214  	    int ret = lstat(profile_1_filename, &st);
215  	
216  	    if (ret >= 0) {
217  	        return 1;
218  	    }
219  	    else {
220  	        return 0;
221  	    }
222  	}

224  	static IARM_Result_t _SetHDCPProfile(void *arg)
225  	{
226  	    IARM_Result_t retStatus = IARM_RESULT_INVALID_STATE;
227  	    IARM_BUS_SYSMGR_HDCPProfileInfo_Param_t *param = (IARM_BUS_SYSMGR_HDCPProfileInfo_Param_t *)arg;
228  	    int new_profile = 0;
229  	
230  		new_profile = param->HdcpProfile;
231  	
232  	    if ((CheckHdcpProfile() != new_profile) && (new_profile == 0 || new_profile == 1)) {
233  	        int fd = -1;
234  	        if (new_profile == 0) {
235  	            unlink(profile_1_filename);
236  	        }
237  	        else {
238  	            fd = open(profile_1_filename, O_WRONLY|O_CREAT, 0666);
239  	            if (fd >= 0) {
240  	                close(fd);
241  	            }
242  	        }
243  	
244  	        {
245  	            __TIMESTAMP(); printf ("The hdcp profile is set to %d\n", new_profile);
246  				IARM_Bus_SYSMgr_EventData_t eventData;
247  				eventData.data.hdcpProfileData.hdcpProfile = new_profile;
248  				__TIMESTAMP(); printf("<<<<<<< Send HDCP Profile UPdate Event is %d>>>>>>>>",eventData.data.hdcpProfileData.hdcpProfile);
249  				retStatus = IARM_Bus_BroadcastEvent(IARM_BUS_SYSMGR_NAME, (IARM_EventId_t)IARM_BUS_SYSMGR_EVENT_HDCP_PROFILE_UPDATE,(void *)&eventData,sizeof(eventData));
250  			}
251  	    }
252  	    return retStatus;
253  	}

282  	{
283  	#ifdef USE_MFR_FOR_SERIAL
284  	    GetSerialNumber();
285  	#endif
286  	    pthread_mutex_lock(&tMutexLock);
287  	    static int current_channel_map_state = 0;
288  	    IARM_Bus_SYSMgr_GetSystemStates_Param_t *param = (IARM_Bus_SYSMgr_GetSystemStates_Param_t *)arg;
289  	
290  	    if(current_channel_map_state != systemStates.channel_map.state) {
291  	      __TIMESTAMP();LOG("_GetSystemStates return ChannelMapState = %d\r\n", systemStates.channel_map.state);
292  	      current_channel_map_state = systemStates.channel_map.state;
293  	    }
294  	
295  	    #ifdef MEDIA_CLIENT
296  	    	systemStates.channel_map.state=2;
297  	    	systemStates.TuneReadyStatus.state=1;
298  		if(systemStates.time_source.state==0) {
299  	           if (access(NTP_FILE_NAME, F_OK ) != -1 ) {
300  	      	      systemStates.time_source.state=1;
301  		   }
302  		}
303  	    #endif
304  	    *param = systemStates;
305  	    pthread_mutex_unlock(&tMutexLock);
306  	    return IARM_RESULT_SUCCESS;
307  	}

309  	static void _sysEventHandler(const char *owner, IARM_EventId_t eventId, void *data, size_t len)
310  	{
311  	    errno_t rc = -1;
312  		/* Only handle state events */
313  	    if (eventId != IARM_BUS_SYSMGR_EVENT_SYSTEMSTATE) return;
314  	
315  		/*Handle only Sys Manager Events */
316  		if (strcmp(owner, IARM_BUS_SYSMGR_NAME)  == 0)
317  		{
318  			//__TIMESTAMP();LOG("_sysEventHandler invoked ww\r\n");
319  			pthread_mutex_lock(&tMutexLock);
320  			IARM_Bus_SYSMgr_EventData_t *sysEventData = (IARM_Bus_SYSMgr_EventData_t*)data;
321  	
322  			IARM_Bus_SYSMgr_SystemState_t stateId = sysEventData->data.systemStates.stateId;
323  			int state = sysEventData->data.systemStates.state;
324  			int error = sysEventData->data.systemStates.error;
325  			char* payload=sysEventData->data.systemStates.payload;
326  			__TIMESTAMP();LOG("_sysEventHandler invoked for stateid %d of state %d and error %d\r\n", stateId, state,error);
327  			switch(stateId) {
328  				case IARM_BUS_SYSMGR_SYSSTATE_CHANNELMAP:
329  					systemStates.channel_map.state = state;
330  					systemStates.channel_map.error = error;
331  					/* memcpy can be replaced with strcpy once the RI is replaced with RMF */
332  						rc = memcpy_s( systemStates.channel_map.payload,sizeof(systemStates.channel_map.payload), payload, sizeof( systemStates.channel_map.payload ) );
333  						if(rc!=EOK)
334  						{
335  							ERR_CHK(rc);
336  						}
337  						memcpy( systemStates.channel_map.payload, payload, sizeof( systemStates.channel_map.payload ) );
338  					systemStates.channel_map.payload[ sizeof( systemStates.channel_map.payload ) - 1] = 0;
339  					LOG( "Got  IARM_BUS_SYSMGR_SYSSTATE_CHANNELMAP ID = %s", systemStates.channel_map.payload );
340  					break;
341  				case IARM_BUS_SYSMGR_SYSSTATE_DISCONNECTMGR:
342  					systemStates.disconnect_mgr_state.state = state;
343  					systemStates.disconnect_mgr_state.error = error;
344  					LOG( "Got  IARM_BUS_SYSMGR_SYSSTATE_DISCONNECTMGR state = %d\n", state );
345  					break;
346  				case IARM_BUS_SYSMGR_SYSSTATE_TUNEREADY:
347  					systemStates.TuneReadyStatus.state = state;
348  					systemStates.TuneReadyStatus.error = error;
349  					break;
350  				case   IARM_BUS_SYSMGR_SYSSTATE_EXIT_OK :
351  					systemStates.exit_ok_key_sequence.state = state;
352  					systemStates.exit_ok_key_sequence.error = error;
353  					break;
354  				case   IARM_BUS_SYSMGR_SYSSTATE_CMAC :

357  					break;
358  				case   IARM_BUS_SYSMGR_SYSSTATE_MOTO_ENTITLEMENT :
359  					systemStates.card_moto_entitlements.state = state;
360  					systemStates.card_moto_entitlements.error = error;
361  					break;
362  				case   IARM_BUS_SYSMGR_SYSSTATE_MOTO_HRV_RX :
363  					systemStates.card_moto_hrv_rx.state = state;

586  	    pthread_mutex_lock(&tMutexLock);

594  	    param.bufLen = 0;

596  	    param.buffer[param.bufLen] = '\0';
597  	
598  	    if(iarm_ret == IARM_RESULT_SUCCESS)
599  	    {
600  	        memset(systemStates.stb_serial_no.payload, 0, sizeof(systemStates.stb_serial_no.payload));
