132  	IARM_Result_t DSMgr_Start()
133  	{
134  		FILE *fDSCtrptr = NULL;
135  		IARM_Bus_SYSMgr_GetSystemStates_Param_t tuneReadyParam;
136  	    IARM_Result_t iarmStatus;
137  	
138  		setvbuf(stdout, NULL, _IOLBF, 0);
139  	    INT_INFO("Entering [%s] - [%s] - disabling io redirect buf \r\n", __FUNCTION__, IARM_BUS_DSMGR_NAME);
140  	
141  		/* Register with IARM Libs and Connect */
142  	    iarmStatus = IARM_Bus_Init(IARM_BUS_DSMGR_NAME);
143  	    if (IARM_RESULT_SUCCESS != iarmStatus) {
144  	        INT_ERROR("Failed to initialize IARM Bus for [%s] \r\n", IARM_BUS_DSMGR_NAME);
145  	        return iarmStatus;
146  	    }
147  	    iarmStatus = IARM_Bus_Connect();
148  	    if (IARM_RESULT_SUCCESS != iarmStatus) {
149  	        INT_ERROR("Failed to connect IARM Bus for [%s] \r\n", IARM_BUS_DSMGR_NAME);
150  	        return iarmStatus;
151  	    }
152  	    iarmStatus = IARM_Bus_RegisterEvent(IARM_BUS_DSMGR_EVENT_MAX);
153  	    if (IARM_RESULT_SUCCESS != iarmStatus) {
154  	        INT_ERROR("Failed to register IARM Bus events for [%s] \r\n", IARM_BUS_DSMGR_NAME);
155  	        return iarmStatus;
156  	    }
157  		/*Initialize the DS Manager - DS Srv and DS HAL */
158  	    iarmStatus = dsMgr_init();
159  	    if (IARM_RESULT_SUCCESS != iarmStatus) {
160  	        INT_ERROR("Failed to initialize DS Manager for [%s] \r\n", IARM_BUS_DSMGR_NAME);
161  	        return iarmStatus;
162  	    }
163  	
164  		iInitResnFlag = 1;
165  	        dsEdidIgnoreParam_t ignoreEdidParam;
166  	        memset(&ignoreEdidParam,0,sizeof(ignoreEdidParam));
167  	        ignoreEdidParam.handle = dsVIDEOPORT_TYPE_HDMI;
168  	        _dsGetIgnoreEDIDStatus(&ignoreEdidParam);
169  		IsIgnoreEdid_gs = ignoreEdidParam.ignoreEDID;
170  		INT_INFO("ResOverride DSMgr_Start IsIgnoreEdid_gs: %d\n", IsIgnoreEdid_gs);
171  		/*Register the Events */
172  		iarmStatus = IARM_Bus_RegisterEventHandler(IARM_BUS_SYSMGR_NAME,IARM_BUS_SYSMGR_EVENT_SYSTEMSTATE,_EventHandler);
173  		if (IARM_RESULT_SUCCESS != iarmStatus) {
174  			INT_ERROR("Failed to register IARM Bus events for [%s] \r\n", IARM_BUS_SYSMGR_NAME);
175  			return iarmStatus;
176  		}
177  		iarmStatus = IARM_Bus_RegisterEventHandler(IARM_BUS_DSMGR_NAME,IARM_BUS_DSMGR_EVENT_HDMI_HOTPLUG,_EventHandler);
178  		if (IARM_RESULT_SUCCESS != iarmStatus) {
179  			INT_ERROR("Failed to register IARM Bus events for [%s] \r\n", IARM_BUS_DSMGR_NAME);
180  			return iarmStatus;
181  		}
182  		iarmStatus = IARM_Bus_RegisterEventHandler(IARM_BUS_DSMGR_NAME,IARM_BUS_DSMGR_EVENT_HDCP_STATUS,_EventHandler);
183  		if (IARM_RESULT_SUCCESS != iarmStatus) {
184  			INT_ERROR("Failed to register IARM Bus events for [%s] \r\n", IARM_BUS_DSMGR_NAME);
185  			return iarmStatus;
186  		}
187  		/*Register EAS handler so that we can ensure audio settings for EAS */
188  		iarmStatus = IARM_Bus_RegisterCall(IARM_BUS_COMMON_API_SysModeChange, _SysModeChange);
189  		if (IARM_RESULT_SUCCESS != iarmStatus) {
190  			INT_ERROR("Failed to register IARM Bus events for [%s] \r\n", IARM_BUS_COMMON_API_SysModeChange);
191  			return iarmStatus;
192  		}
193  	        /*Refactored dsMGR code*/
194  	       PowerController_Init();
195  	       dsMgrInitPwrControllerEvt();
196  	       /* Power controller connect is checked inside initPwrEventListner*/
197  	        initPwrEventListner();
198  		/* Create  Thread for listening Hot Plug events */
199  		pthread_mutex_init (&tdsMutexLock, NULL);
200  		if (pthread_cond_init(&tdsMutexCond, NULL) != 0) {
201  			INT_ERROR("Failed to create pthread_cond_init tdsMutexCond.");
202  			return IARM_RESULT_IPCCORE_FAIL;
203  		}
204  		if (pthread_create(&edsHDMIHPDThreadID, NULL, _DSMgrResnThreadFunc, NULL) != 0) {
205  			INT_ERROR("Failed pthread_create _DSMgrResnThreadFunc.");
206  			return IARM_RESULT_IPCCORE_FAIL;
207  		}
208  	
209  		/* Read the HDMI DDC Line delay to be introduced
210  		 * for setting  the resolution
211  		 * The DDC line is used for EDID and HDCP Negotiation
212  		 */
213  		fDSCtrptr = fopen("/opt/ddcDelay","r");
214  		if (NULL != fDSCtrptr)
215  		{
216  			if(0 > fscanf(fDSCtrptr,"%d",&iResnCount))
217  			{
218  				INT_ERROR("Error: fscanf on ddcDelay failed");
219  			}
220  	                fclose (fDSCtrptr);
221  		}
222  		INT_DEBUG("Retry DS manager Resolution count is iResnCount = %d \r\n",iResnCount);
223  	
224  	
225  		iarmStatus = IARM_Bus_Call(IARM_BUS_SYSMGR_NAME, IARM_BUS_SYSMGR_API_GetSystemStates, &tuneReadyParam, sizeof(tuneReadyParam));
226  		if (IARM_RESULT_SUCCESS != iarmStatus) {
227  			INT_ERROR("Failed to get Tune Ready status for [%s] \r\n", IARM_BUS_SYSMGR_NAME);
228  			return iarmStatus;
229  		}
230  		INT_INFO("Tune Ready Status on start up is %d \r\n",tuneReadyParam.TuneReadyStatus.state);
231  	
232  		/* Read the Tune Ready status on startup  and update the tune ready flag */
233  		if(1 == tuneReadyParam.TuneReadyStatus.state)
234  		{
235  			iTuneReady = 1;
236  		}
237  	
238  		/* Create Main loop for DS Manager */
239  	    dsMgr_Gloop = g_main_loop_new ( NULL , FALSE );
240  	    if(dsMgr_Gloop != NULL){
241  	        g_timeout_add_seconds (300 , heartbeatMsg , dsMgr_Gloop); 
242  	    }
243  	    else {
244  	        INT_ERROR("Fails to Create a main Loop for [%s] \r\n",IARM_BUS_DSMGR_NAME);
245  	    }
246  	
247  	    INT_INFO("Set resolution during dsMgr init .. \r\n");
248  	    _SetVideoPortResolution(); 
249  	    return IARM_RESULT_SUCCESS;
250  	}

252  	IARM_Result_t DSMgr_Loop()
253  	{
254  	   
255  	   /* DS Mgr loop */
256  	    if(dsMgr_Gloop)
257  	    { 
258  	        g_main_loop_run (dsMgr_Gloop);
259  	        g_main_loop_unref(dsMgr_Gloop);
260  	    }
261  		return IARM_RESULT_SUCCESS;
262  	}

264  	static gboolean heartbeatMsg(gpointer data)
265  	{
266  	    time_t curr = 0;
267  	    time(&curr);
268  	    INT_INFO("I-ARM BUS DS Mgr: HeartBeat at %s\r\n", ctime(&curr));
269  	    return TRUE;
270  	}
271  	
272  	IARM_Result_t DSMgr_Stop()
273  	{
274  	    IARM_Result_t iarmStatus = IARM_RESULT_SUCCESS;
275  		if(dsMgr_Gloop)
276  	    {
277  	        g_main_loop_quit(dsMgr_Gloop);
278  	    }
279  	    dsMgrDeinitPwrControllerEvt();
280  	    PowerController_Term();
281  	    iarmStatus = IARM_Bus_Disconnect();
282  	    if (IARM_RESULT_SUCCESS != iarmStatus) {
283  	        INT_ERROR("Failed to disconnect IARM Bus for [%s] \r\n", IARM_BUS_DSMGR_NAME);
284  	        return iarmStatus;
285  	    }
286  	    iarmStatus = IARM_Bus_Term();
287  	    if (IARM_RESULT_SUCCESS != iarmStatus) {
288  	        INT_ERROR("Failed to terminate IARM Bus for [%s] \r\n", IARM_BUS_DSMGR_NAME);
289  	        return iarmStatus;
290  	    }
291  		pthread_mutex_destroy (&tdsMutexLock);
292  		pthread_cond_destroy  (&tdsMutexCond);
293  	
294  	    return IARM_RESULT_SUCCESS;
295  	}

297  	/**
298  	 * @fn static IARM_Result_t _SysModeChange(void *arg){
299  	 * @brief This function is a event handler which returns current system
300  	 *  mode using IARM. It returns mode as  "NORMAL", "WAREHOUSE","EAS" or "UNKNOWN".
301  	 *
302  	 * @param[in] void pointer to void, containing IARM_Bus_CommonAPI_SysModeChange_Param_t data.
303  	 *
304  	 * @return variable of IARM_Result_t type.
305  	 * @retval IARM_RESULT_SUCCESS On function completion.
306  	 */
307  	static IARM_Result_t _SysModeChange(void *arg)
308  	{
309  	    IARM_Bus_CommonAPI_SysModeChange_Param_t *param = (IARM_Bus_CommonAPI_SysModeChange_Param_t *)arg;
310  	    IARM_Bus_Daemon_SysMode_t isNextEAS = IARM_BUS_SYS_MODE_NORMAL;
311  	
312  	    INT_INFO("[DSMgr] Recvd Sysmode Change::New mode --> %d,Old mode --> %d",param->newMode,param->oldMode);
313  	         
314  	     if ((param->newMode == IARM_BUS_SYS_MODE_EAS) ||  
315  	     	 (param->newMode == IARM_BUS_SYS_MODE_NORMAL)) 
316  	     {
317  	        isNextEAS = param->newMode;
318  	     }
319  	     else
320  	     {
321  	     	/* Do not process any other mode change as of now for DS Manager.. */
322  	     	return IARM_RESULT_SUCCESS;
323  	     }
324  	
325  	    if ((isEAS == IARM_BUS_SYS_MODE_EAS) && (isNextEAS == IARM_BUS_SYS_MODE_NORMAL) ) {
326  	        isEAS = IARM_BUS_SYS_MODE_NORMAL;
327  	        _setAudioMode();
328  	    }
329  	    else if ((isEAS == IARM_BUS_SYS_MODE_NORMAL) && (isNextEAS == IARM_BUS_SYS_MODE_EAS) ) {
330  	    	/* 
331  	    		* Change the Audio Mode to Stereo if Current Audio Setting is Paasthrough
332  	    	*/
333  	        isEAS = IARM_BUS_SYS_MODE_EAS;
334  	        _setEASAudioMode();
335  	        
336  	    }
337  	    else {
338  	        /* no op for no mode change */
339  	    }
340  	    return IARM_RESULT_SUCCESS;
341  	}

363  	/*Event Handler for DS Manager And Sys Manager Events */
364  	static void _EventHandler(const char *owner, IARM_EventId_t eventId, void *data, size_t len)
365  	{
366  	    /* allows dsmgr to set initial resolution irrespective of ignore edid only during boot */
367  	    static bool bootup_flag_enabled = true;
368  	
369  	    /*Handle only Sys Manager Events */
370  		if (strcmp(owner, IARM_BUS_SYSMGR_NAME)  == 0) 
371  		{
372  			/* Only handle state events */
373  		    if (eventId != IARM_BUS_SYSMGR_EVENT_SYSTEMSTATE) return;
374  			//	INT_DEBUG("_sysMgrEventHandler invoked in DS Manager\r\n");
375  			IARM_Bus_SYSMgr_EventData_t *sysEventData = (IARM_Bus_SYSMgr_EventData_t*)data;
376  			IARM_Bus_SYSMgr_SystemState_t stateId = sysEventData->data.systemStates.stateId;
377  			int state = sysEventData->data.systemStates.state;
378  			INT_INFO("_sysEventHandler invoked for stateid %d of state %d \r\n", stateId, state);
379  			switch(stateId) {
380  				case IARM_BUS_SYSMGR_SYSSTATE_TUNEREADY:
381  						INT_DEBUG("Tune Ready Evenets in DS Manager \r\n");
382  					
383  						if (0 == iTuneReady)
384  						{
385  							iTuneReady = 1;
386  	
387  							/* Set audio mode from persistent */
388  							_setAudioMode();
389  							
390  							/* Un-block the Resolution Settings Thread */
391  							pthread_mutex_lock(&tdsMutexLock);
392  							pthread_cond_signal(&tdsMutexCond);
393  							pthread_mutex_unlock(&tdsMutexLock);
394  						}
395  	  			break;
396  				default:
397  					break;
398  			}
399  		}else if (strcmp(owner,IARM_BUS_DSMGR_NAME) == 0)
400  	    {
401  	        switch (eventId) {
402  				case IARM_BUS_DSMGR_EVENT_HDMI_HOTPLUG:
403  					{
404  						
405  						IARM_Bus_DSMgr_EventData_t *eventData = (IARM_Bus_DSMgr_EventData_t *)data;
406  						
407  						INT_INFO("[DsMgr] Got HDMI %s Event  \r\n",(eventData->data.hdmi_hpd.event == dsDISPLAY_EVENT_CONNECTED ? "Connect" : "Disconnect"));
408  	
409  	                                        setBGColor(dsVIDEO_BGCOLOR_NONE);
410  	
411  						/* Un-Block the Resolution Settings Thread */
412  						pthread_mutex_lock(&tdsMutexLock);
413  						edisplayEventStatus = ((eventData->data.hdmi_hpd.event == dsDISPLAY_EVENT_CONNECTED) ? dsDISPLAY_EVENT_CONNECTED : dsDISPLAY_EVENT_DISCONNECTED);
414  						pthread_cond_signal(&tdsMutexCond);
415  						pthread_mutex_unlock(&tdsMutexLock);
416  											
417  					}
418  	                break;
419  				case IARM_BUS_DSMGR_EVENT_HDCP_STATUS:
420  					{
421  						IARM_Bus_DSMgr_EventData_t *eventData = (IARM_Bus_DSMgr_EventData_t *)data;
422  						IARM_Bus_SYSMgr_EventData_t HDCPeventData;
423  						int status = eventData->data.hdmi_hdcp.hdcpStatus;
424  						//INT_DEBUG("%s: IARM_BUS_DSMGR_EVENT_HDCP_STATUS  event status :%d \r\n",__FUNCTION__, status);
425  						
426  						/* HDCP is enabled */
427  						HDCPeventData.data.systemStates.stateId = IARM_BUS_SYSMGR_SYSSTATE_HDCP_ENABLED;
428  						HDCPeventData.data.systemStates.state = 1;
429  						if (status == dsHDCP_STATUS_AUTHENTICATED )
430  						{
431  							INT_INFO("Changed status to HDCP Authentication Pass  !!!!!!!! ..\r\n");
432  							HDCPeventData.data.systemStates.state =  1;
433  	                                                bHDCPAuthenticated = true;
434  							INT_INFO("HDCP success - Removed hotplug_event_src Time source %d and set resolution immediately \r\n",hotplug_event_src);
435  	                                                if(hotplug_event_src)
436  	                                                {
437  	                                                    g_source_remove(hotplug_event_src);
438  	                                                    hotplug_event_src = 0;
439  	                                                }
440  	                                                setBGColor(dsVIDEO_BGCOLOR_NONE);
441  							if ((!IsIgnoreEdid_gs) || bootup_flag_enabled){
442  	                                                    _SetVideoPortResolution();
443  							    if(bootup_flag_enabled)
444  							        bootup_flag_enabled = false;
445  							}
446  	                                                g_timeout_add_seconds((guint)1,dumpEdidOnChecksumDiff,NULL);
447  						} 
448  						else if (status == dsHDCP_STATUS_AUTHENTICATIONFAILURE )
449  						{
450  							INT_ERROR("Changed status to HDCP Authentication Fail   !!!!!!!! ..\r\n");
451  							HDCPeventData.data.systemStates.state =  0;
452  							
453  							bHDCPAuthenticated = false;
454  							if (!IsIgnoreEdid_gs) {
455  	                                                    _SetVideoPortResolution();
456  							}
457  	                                                g_timeout_add_seconds((guint)1,dumpEdidOnChecksumDiff,NULL);
458  	                    }
459  	
460  	                    if (IARM_Bus_BroadcastEvent(IARM_BUS_SYSMGR_NAME, (IARM_EventId_t)IARM_BUS_SYSMGR_EVENT_SYSTEMSTATE,
461  	                                (void *)&HDCPeventData, sizeof(HDCPeventData)) != IARM_RESULT_SUCCESS) {
462  	                        INT_ERROR("Failed to broadcast HDCP status for [%s] \r\n", IARM_BUS_SYSMGR_NAME);
463  	                    }
464  					}
465  	                break;	
466  				default:
467  					break;
468  				}
469  		}
470  	}

472  	/* Set Video resolution on HDMI Hot Plug or Tune Ready events  */
473  	static int _SetVideoPortResolution()
474  	{
475  		intptr_t _hdmihandle = 0;
476  		intptr_t _comphandle = 0;
477  		bool connected=false;
478  		int iCount = 0;
479  		
480  	
481  		INT_DEBUG("%s:Enter \r\n",__FUNCTION__);
482  		 
483  	        _hdmihandle = getVideoPortHandle(dsVIDEOPORT_TYPE_HDMI);	
484  		if(_hdmihandle != NULL)
485  		{		
486  	
487  			usleep(100*1000);	//wait for 100 milli seconds
488  	
489  	
490  			/* 	
491  				* Check for HDMI DDC Line  when HDMI is connected.
492  			*/
493  	                connected = isHDMIConnected();
494  			if(iInitResnFlag && connected)
495  			{
496  	
497  				#ifdef _INIT_RESN_SETTINGS  	
498  					/*Wait for iResnCount*/
499  					while(iCount < iResnCount)
500  					{
501  						sleep(1);	//wait for 1 sec
502  						if (dsGetHDMIDDCLineStatus())
503  						{
504  							break;
505  						}
506  						INT_INFO ("Waiting for HDMI DDC Line to be ready for resolution Change...\r\n");
507  						iCount++;
508  					}
509  				#endif
510  			}
511

563  	/**
564  	 * @brief This Function does following :
565  	 *   Read Persisted resolution 
566  	 *   Verify Persisted resolution with Platform and EDID resolution 
567  	 *   If fails set best EDID resolution supported by platform 
568  	 *   If fails Default to 720P
569  	 * 	 If 720p is not supported by TV , Default to 480p
570  	 *  @param void pointer Device Handle
571  	 * * @param Connection Status 
572  	 ** @return void pointer (NULL)
573  	 */
574  	static int  _SetResolution(intptr_t* handle,dsVideoPortType_t PortType)
575  	{
576  		errno_t rc = -1;
577  		intptr_t _displayHandle = 0;
578  		int numResolutions = 0,i=0;
579  		intptr_t _handle = *handle;
580  		bool IsValidResolution = false;
581  		dsVideoPortSetResolutionParam_t Setparam;
582  		dsVideoPortGetResolutionParam_t Getparam;
583  		dsVideoPortResolution_t *setResn = NULL;
584  		dsDisplayEDID_t edidData;
(1) Event stack_use_local_overflow: Local variable "Edidparam" uses 13360 bytes of stack space, which exceeds the maximum single use of 10000 bytes. 
585  		dsDisplayGetEDIDParam_t Edidparam;
586  		/*
587  			* Default Resolution Compatible check is false - Do not Force compatible resolution on startup
588  		*/
589  		Setparam.forceCompatible = false;
590  	
591  		/*Initialize the struct*/
592  		memset(&edidData, 0, sizeof(edidData));
593  		
594  		/* Return if Handle is NULL */
595  		if (_handle == NULL)
596  		{
597  			INT_ERROR("_SetResolution - Got NULL Handle ..\r\n");
598  			return 0;
599  		}
600  		
601  		/*Get the User Persisted Resolution Based on Handle */
602  		memset(&Getparam,0,sizeof(Getparam));
603  		Getparam.handle = _handle;
604  		Getparam.toPersist = true;
605  		_dsGetResolution(&Getparam);
606  		dsVideoPortResolution_t *presolution = &Getparam.resolution;

609

619  				rc = memcpy_s(&edidData,sizeof(edidData), &Edidparam.edid, sizeof(Edidparam.edid));

622  					ERR_CHK(rc);
623  				}

625  				numResolutions = edidData.numOfSupportedResolution;
626  				INT_INFO("numResolutions is %d \r\n",numResolutions);
627  				
628  				/*  If HDMI is connected and Low power Mode. 
629  					The TV might not Transmit the EDID information 
630  					Change the Resolution in Next Hot plug
631  					DO not set the Resolution if TV is in DVI mode.
632  				*/
633  				if ((0 == numResolutions) || (!(edidData.hdmiDeviceType)))
634  				{
635  	
636  					INT_ERROR("Do not Set Resolution..The HDMI is not Ready  !! \r\n");
637  					INT_ERROR("numResolutions  = %d edidData.hdmiDeviceType = %d !! \r\n",numResolutions,edidData.hdmiDeviceType);
638  					return 0;
639  				}
640  				
641  				/* 
642  					* Check if Persisted Resolution matches with 
643  					* TV Resolution  list	
644  				*/
645  				for (i = 0; i < numResolutions; i++)
646  				{
647  					setResn = &(edidData.suppResolutionList[i]);
648  					INT_INFO("presolution->name : %s, resolution->name : %s\r\n",presolution->name,setResn->name);
649  					if ((strcmp(presolution->name,setResn->name) == 0 ))
650  					{
651  						INT_INFO("Breaking..Got Platform Resolution - %s..\r\n",setResn->name);
652  						IsValidResolution = true;
653  						Setparam.forceCompatible = true;
654  						break;
655  					}
656  				}
657  	                        /*
658  	                        * The Persisted Resolution Does not matches with TV resolution list 
659  				*  Set  the Best Resolution Supported by TV and Platform
660  				*/
661  	                        if (false == IsValidResolution)
662  				{
663  	                        	/* Set  the Best Resolution Supported by TV and Platform*/
664  	                		for (i = numResolutions-1; i >= 0; i--)
665  	                		{
666  	                        		setResn = &(edidData.suppResolutionList[i]);
667  	                        		int pNumResolutions = dsUTL_DIM(kResolutions);
668  	                        		for (int j = pNumResolutions-1; j >=0; j--)
669  			        		{
670  			                		dsVideoPortResolution_t *pfResolution = &kResolutions[j];
671  			                		if (0 == (strcmp(pfResolution->name,setResn->name)))
672  			                		{

674  			                			IsValidResolution = true;
675  			                    			break;
676  			                		}
677  	                                        }
678  	                                	if (IsValidResolution)
679  	                                	{
680  	                                		break;
681  	                                        }
682  	                                }
683  	                        }
684  				/* 
685  					* The Persisted Resolution Does not matches with TV and Platform 
686  					* Resolution List
687  					* Force PLatform Default Resolution 	
688  				*/
689  				if (false == IsValidResolution)
690  				{
691  					/* Check if the Default platform resolution is supported by Platfrom resolution List i.e 720p */
692  					dsVideoPortResolution_t *defaultResn; 
693  					defaultResn = &kResolutions[kDefaultResIndex];
694  					for (i = 0; i < numResolutions; i++)
695  					{
696  						setResn = &(edidData.suppResolutionList[i]);
697  						//INT_DEBUG("\n presolution->name : %s, resolution->name : %s\n",defaultResn->name,setResn->name);
698  						if ((strcmp(defaultResn->name,setResn->name) == 0 ))
699  						{
700  							IsValidResolution = true;
701  							INT_INFO("Breaking..Got Default Platform Resolution - %s..\r\n",setResn->name);
702  							break;
703  						}
704  					}
705  				}
706  	
707  				if (false == IsValidResolution)
708  				{
709  					/*Take 480p as resolution if both above cases fail */
710  	                for (i = 0; i < numResolutions; i++)
711  	                {
712  	                    setResn = &(edidData.suppResolutionList[i]);
713  	                    if ((strcmp("480p",setResn->name) == 0 )) 
714  	                	{
715  							INT_INFO("Breaking..Default to 480p Resolution - %s..\r\n",setResn->name);
716  							IsValidResolution = true;
717  							break;
718  		               	}
719  					}
720  				}
721  	
722  				if (false == IsValidResolution)
723  				{
724  					/* Boot with  the Resolution Supported by TV and Platform*/
725  	                for (i = 0; i < numResolutions; i++)
726  	                {
727  	                    setResn = &(edidData.suppResolutionList[i]);
728  	                    size_t numResolutions = dsUTL_DIM(kResolutions);
729  	                    for (size_t j = 0; j < numResolutions; j++)
730  			            {
731  			                dsVideoPortResolution_t *pfResolution = &kResolutions[j];
732  			                if (0 == (strcmp(pfResolution->name,setResn->name)))
733  			                {
734  			                    INT_INFO("[DsMgr] Boot with TV Supported Resolution %s \r\n",pfResolution->name);
735  			                    IsValidResolution = true;
736  			                    break;
737  			                }    
738  			            }
739  					}
740  				}
741  			}
742  		}
743  		else if (PortType == dsVIDEOPORT_TYPE_COMPONENT || PortType == dsVIDEOPORT_TYPE_BB || PortType == dsVIDEOPORT_TYPE_RF)
744  		{
745  			/* Set the Component / Composite  Resolution */	
746  			numResolutions = dsUTL_DIM(kResolutions);
747  	    	for (i = 0; i < numResolutions; i++)
748  	    	{
749  	    		setResn = &kResolutions[i];
750  	    		if ((strcmp(presolution->name,setResn->name) == 0 ))
751  	    		{
752  					INT_INFO("Breaking..Got Platform Resolution - %s..\r\n",setResn->name);
753  	        		IsValidResolution = true;
754  	        		break;
755  	    		}
756  	    	}
757  		}
758  			/*  If the Persisted Resolution settings does not matches with Platform Resolution - 
759  				Force Default on Component/Composite 
760  				This is to keep upward compatible and if we intend to 
761  				remove any resolution from Dynamic Resolution List
762  			*/
763  		if(false == IsValidResolution)
764  		{
765  			setResn = &kResolutions[kDefaultResIndex];
766  		}
767  		
768  		/* Set The Video Port Resolution in Requested Handle */
769  		Setparam.handle = _handle;
770  		Setparam.toPersist = false;
771  		
772  		/* If 4K support is disabled and last known resolution is 4K, default to 720p (aka default resolution) */
773  		dsForceDisable4KParam_t res_4K_override;
774  		memset(&res_4K_override, 0, sizeof(res_4K_override));
775  		_dsGetForceDisable4K((void *) &res_4K_override);
776  		if(true == res_4K_override.disable)
777  		{
778  			if(0 == strncmp(presolution->name, "2160", 4))

780  				INT_INFO("User persisted 4K resolution. Now limiting to default (720p?) as 4K support is now disabled.\n");
781  				setResn = &kResolutions[kDefaultResIndex];

784  		
785  		Setparam.resolution = *setResn;

788  		#ifdef _INIT_RESN_SETTINGS

792  				_dsInitResolution(&Setparam);
793  				return 0 ;
794  			}
795  		#endif
796  	
797  		_dsSetResolution(&Setparam);
798  		return 0 ;
799  	}

802  	 * @brief Thread entry fuction to post Resolution on Hot Plug and Tune ready Events
803  	 *
804  	 * This functions changes the device resolution on Hot Plug and Tune ready Events
805  	 *
806  	 * @param void pointer (NULL)
807  	 *
808  	 * @return void pointer (NULL)
809  	 */
810  	static void* _DSMgrResnThreadFunc(void *arg)
811  	{
812  	   
813  		/* Loop */
814  	    while (1)
815  	    {
816  			 INT_INFO ("_DSMgrResnThreadFunc... wait for for HDMI or Tune Ready Events \r\n");
817  			
818  			/*Wait for the Event*/
819  			pthread_mutex_lock(&tdsMutexLock);
820  			pthread_cond_wait(&tdsMutexCond, &tdsMutexLock);
821  			pthread_mutex_unlock(&tdsMutexLock);
822  		  
823  		    INT_INFO("%s: Setting Resolution On:: HDMI %s Event  with TuneReady status = %d \r\n",__FUNCTION__, (edisplayEventStatus == dsDISPLAY_EVENT_CONNECTED ? "Connect" : "Disconnect"),iTuneReady);
824  	
825  	
826  			//On hot plug event , Remove event source 
827  			if(hotplug_event_src)
828  			{
829  				g_source_remove(hotplug_event_src);
830  				INT_INFO("Removed Hot Plug Event Time source %d  \r\n",hotplug_event_src);
831  	      hotplug_event_src = 0;
832  			}
833  	
834  			/*Set the Resolution only on HDMI Hot plug Connect and Tune Ready events */
835  			if((1 == iTuneReady) && (dsDISPLAY_EVENT_CONNECTED == edisplayEventStatus)) {
836  				/*Set Video Output Port  Resolution */
837  	                        if(bHDCPAuthenticated)
838  	                        {
839  	                            _SetVideoPortResolution();
840  				}
841  	                        /* Set audio mode on HDMI hot plug */
842  				_setAudioMode();	
843  			}/*Set the Resolution only on HDMI Hot plug - Disconnect and Tune Ready event */
844  			else if((1 == iTuneReady) && (dsDISPLAY_EVENT_DISCONNECTED == edisplayEventStatus)) {
845  				 /* * To avoid reoslution settings of HDMI hot plug when TV goes from power OFF to ON condition 
846  				 	* Delay the setting of resolution by 5 sec. This will help to filter out un-necessary 
847  				 	* resolution settings on HDMI hot plug.  
848  				 */
849  		             bHDCPAuthenticated = false;
850  	                     if(isComponentPortPresent())
851  	                     {
852  				 hotplug_event_src = g_timeout_add_seconds((guint)5,_SetResolutionHandler,dsMgr_Gloop); 
853  				 INT_INFO("Schedule a handler to set the resolution after 5 sec for %d time src.. \r\n",hotplug_event_src);
854  	                     }
855  			}
856  	 
857  	     }
858  	    return arg;
859  	}

1035 	        INT_INFO("dumpEdidOnChecksumDiff HDMI-EDID Dump>>>>>>>>>>>>>>\r\n");
1036 	        intptr_t _displayHandle = 0;
1037 	        dsGetDisplay(dsVIDEOPORT_TYPE_HDMI, 0, &_displayHandle);
1038 	        if (_displayHandle) {
1039 	                int length = 0;
1040 	                dsDisplayGetEDIDBytesParam_t EdidBytesParam;
1041 	                static int cached_EDID_checksum = 0;
1042 	                int current_EDID_checksum = 0;
1043 	                memset(&EdidBytesParam,0,sizeof(EdidBytesParam));
1044 	                EdidBytesParam.handle = _displayHandle;
1045 	                _dsGetEDIDBytes(&EdidBytesParam);
1046 			length = EdidBytesParam.length;
1047 	
1048 			if((length > 0) && (length <= 512)) {
1049 	                    unsigned char* edidBytes = EdidBytesParam.bytes;
1050 	                    for (int i = 0; i < (length / 128); i++)
1051 	                            current_EDID_checksum += edidBytes[(i+1)*128 - 1];
